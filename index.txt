    <script>
        const startBtn = document.getElementById('startBtn');
        const noteDisplay = document.getElementById('note');
        const freqDisplay = document.getElementById('freq');
        const volBar = document.getElementById('volume-bar');
        const statusText = document.getElementById('status');

        const noteStrings = ["C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B"];

        async function initAudio() {
            try {
                const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                const audioContext = new (window.AudioContext || window.webkitAudioContext)();
                const analyser = audioContext.createAnalyser();
                const source = audioContext.createMediaStreamSource(stream);
                
                analyser.fftSize = 2048;
                source.connect(analyser);

                const bufferLength = analyser.frequencyBinCount;
                const dataArray = new Float32Array(analyser.fftSize);

                statusText.innerText = "正在收音中... 請吹奏！";
                startBtn.style.display = "none";

                function update() {
                    analyser.getFloatTimeDomainData(dataArray);
                    
                    // 1. 計算音量 (RMS) 確保有收到聲音
                    let sum = 0;
                    for (let i = 0; i < dataArray.length; i++) {
                        sum += dataArray[i] * dataArray[i];
                    }
                    let rms = Math.sqrt(sum / dataArray.length);
                    volBar.style.width = (rms * 500) + "%"; // 顯示音量條

                    // 2. 簡單的自相關算法 (Autocorrelation) 找出頻率
                    if (rms > 0.05) { // 只有當音量夠大時才偵測
                        let pitch = autoCorrelate(dataArray, audioContext.sampleRate);
                        if (pitch !== -1) {
                            freqDisplay.innerText = Math.round(pitch) + " Hz";
                            noteDisplay.innerText = getNote(pitch);
                        }
                    } else {
                        noteDisplay.innerText = "--";
                    }

                    requestAnimationFrame(update);
                }
                update();

            } catch (err) {
                statusText.innerText = "錯誤: 無法存取麥克風 (" + err.message + ")";
                console.error(err);
            }
        }

        function autoCorrelate(buf, sampleRate) {
            let SIZE = buf.length;
            let rms = 0;
            for (let i=0; i<SIZE; i++) {
                let val = buf[i];
                rms += val*val;
            }
            if (Math.sqrt(rms/SIZE) < 0.01) return -1;

            let r1=0, r2=SIZE-1, thres=0.2;
            for (let i=0; i<SIZE/2; i++) if (Math.abs(buf[i]) < thres) { r1=i; break; }
            for (let i=1; i<SIZE/2; i++) if (Math.abs(buf[SIZE-i]) < thres) { r2=SIZE-i; break; }
            buf = buf.slice(r1,r2);
            SIZE = buf.length;

            let c = new Array(SIZE).fill(0);
            for (let i=0; i<SIZE; i++)
                for (let j=0; j<SIZE-i; j++)
                    c[i] = c[i] + buf[j]*buf[j+i];

            let d=0; while (c[d]>c[d+1]) d++;
            let maxval=-1, maxpos=-1;
            for (let i=d; i<SIZE; i++) {
                if (c[i] > maxval) {
                    maxval = c[i];
                    maxpos = i;
                }
            }
            let T0 = maxpos;
            return sampleRate/T0;
        }

        function getNote(frequency) {
            const n = 12 * (Math.log2(frequency / 440)) + 69;
            const noteIndex = Math.round(n) % 12;
            return noteStrings[noteIndex];
        }

        startBtn.addEventListener('click', () => {
            initAudio();
        });
    </script>
